export const getActiveMeeting: RequestHandler = async (req, res) => {
  const { employeeId, followUpId } = req.query;
  
  const query: any = {
    status: { $in: ["in-progress", "started"] }
  };
  
  if (followUpId) query.followUpId = followUpId;
  else if (employeeId) query.employeeId = employeeId;
  
  const meeting = await Meeting.findOne(query).sort({ startTime: -1 });
  
  if (!meeting) {
    return res.status(404).json({ error: "No active meeting found" });
  }
  
  res.json(meeting);
};
